package main

// Go语言的非侵入式接口

/*
什么是接口
接口: 只提供了公开的方法, 并没有提供方法的具体实现

MyInterface   getValue
MyClass1      getValue
MyClass2      getValue

多态
非侵入式接口
侵入式接口: 类需要显式指定实现哪一个或哪几个接口
*/

type ByteCounter int

func (c *ByteCounter) Write (p []byte) (n int, err error) {
    *c += (ByteCounter(len(p)))
    n = len(p)
    err = nil
    return
}

}
