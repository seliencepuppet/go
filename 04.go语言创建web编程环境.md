
```go
package main
import (
    "net/http"
    "fmt"
)

// 简单的web服务器
func main() {
    // 1.需要监听客户端的请求
    // 2.响应客户端请求的回调函数
    
    http.HandleFunc(pattern:"/", handle)
    http.ListenAndServer(addr:"localhost:8000", handle: nil)
}

func handle(w http.ResponseWriter, r *http.Request) {
    fmt.Printf(w, format:"URL.Path = %q\n", r.URL.Path)
}
```
