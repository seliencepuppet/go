
```go
package main

import "fmt"

// 封装
// 面向对象3大特性: 继承, 封装和组合
// 一个对象的变量和方法如果对调用方法是不可见的, 就称为封装
// public private


type Enclosure struct {
    Name string
    value int
}

func (self * Enclosure) getValue() int {
    return self.value
}

func (self * Enclosure) getName() string {
    return self.Name
}

func main() {
    e1 := Enclosure{"Bill", 30}
    fmt.Println(e1.Name)
    fmt.Println(e1.value)
}
```
