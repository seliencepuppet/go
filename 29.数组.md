数组

2. 枚举数组中的元素
3. 数组的比较

var arrayName[length] type 
*/

func main() {
    // 初始化数组
    var a[10] int

    // 声明一个固定的数组
    var a1[3] int = [3]int{4, 5, 6}

    for i := range a (
        a[i] = i * i
    )
    
    for i, v := range a (
        fmt.Println("%d: %d\n", i, v)
    )
}
