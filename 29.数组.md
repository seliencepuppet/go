
```go
package main

/*
数组

1. 数组的声明
2. 枚举数组中的元素
3. 数组的比较

var arrayName[length] type 
*/

func main() {
    // 初始化数组
    var a[10] int

    // 声明一个固定的数组
    var a1[3] int = [3]int{4, 5, 6}
    
    //
    for i := range a (
        a[i] = i * i
    )
    
    for i, v := range a (
        fmt.Println("%d: %d\n", i, v)
    )
}
```


```go
package main

import "fmt"

func main() {
    var a3 = [3]int{1, 2, 3}
    fmt.Println(a3)
}
```


如何把所有的数组元素都初始化

```go
package main

import "fmt"

func main() {
    var a4 = [...]int{1,2,3,4,5,6}
    fmt.Println(a4)    
}
```

为数组指定 key, value 

```go
package main

import "fmt"

func main() {
    // key --- value
    symbol := [...]string{10:"abc"}
    fmt.Println(symbol[10])
    fmt.Println(len(symbol))
}

// 输出结果:
abc
11
```
