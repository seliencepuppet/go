package main

import "fmt"
/*
结构体

结构体是一种聚合的数据类型, 是由零个或任意多个不同类型的变量聚合而成的实体, 每个变量成为结构体的成员
经典案例: 表示公司员工的信息
*/

func main() {
    type Employee struct {
        ID int
        name string
        address string
        Position string
        Salary int
    }
    
    var bil = Employee
    bill.ID = 10
    bill.name = "Bill"
    bill.address = "地球"
    bill.Salary = 20000
    bill.Position = "工程师"
    fmt.Println(bill)
    
    position = &bill.Position
    *position = "程序员"
    fmt.Println(bill)
    
    var employee *Employee = &bill
    employee.address = "泰坦"
    fmt.Println(*employee)
}
