
```go
package main
import "fmt"

/*
string - 字符串
不可改变的字节序列
字符串通常会被解释为utf-8编码
len函数获取字符串中字节的长度
字符串可以用下标来进行操作: 0 < i < len(s)

go语言中, 字符串表达用双引号, 字符表示用单引号
*/

func main() {
    s := "hello world"
    fmt.Println(len(s))
    fmt.Println(s[0], s[2])
    fmt.Println(s[0] == "h")

    s1 := "hello 张三"
    fmt.Println(len(s1))
    fmt.Println(s1[10])
}
```
