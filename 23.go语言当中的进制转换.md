
将十进制的数字, 转换成为二进制, 八进制, 十六进制

将数字转换成为二进制

```go
package main
import (
    "fmt"
    "reflect"
)

func main() {
    x := 123
    bin := fmt.Println(strconv.FormatInt(int64(x), 2))
    fmt.Println(bin)
    fmt.Println(reflect.TypeOf(bin))
}

// 输出结果为:
1111011
string
```

将数字转换成为八进制

```go
package main
import (
    "fmt"
    "reflect"
)

func main() {
    x := 123
    oct := fmt.Println(strconv.FormatInt(int64(x), 8))
    fmt.Println(oct)
    fmt.Println(reflect.TypeOf(oct))
}

// 输出结果为:

```
