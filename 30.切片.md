切片是获取数组当中的子数组

```go
package main

import "fmt

func main() {
    numbers := []int{44,3,1,67,34}
    n1 := numbers[1:3]
    fmt.Println(n1)
    n1[1] = 200
    fmt.Println(n1)
    fmt.Println(numbers)
    
    n2 := numbers[2:len(numbers)]
    n3 := numbers[2:]
    fmt.Println(n2)
    fmt.Println(n3)
}

// 输出结果如下:
[3 1]
[3 200]
[44 3 200 67 34]
```
