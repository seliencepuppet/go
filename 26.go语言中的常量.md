
```go
package main

/*
常量

常量表达式的值是在编译器进行计算的, 而不是运行时, 为了减少运行时错误, 也为了编译优化
const 
常量类型必须是基础类型, 比如: 布尔, 数字, string
*/
func main() {
    // 定义一个常量
    const pi = 3.1415926
    
    // 定义多个常量
    const = (
        e = 2.176898
        name = "Mike"
    )
    
    // 声明数字常量
    var = (
        x = 10
        y = 20
    )
    
    // 声明一个数组常量
    var names[10] string
    fmt.Println(names)
}
```
