go 语言中函数必须以 func 开头

```go
package main

import "fmt"

/*
// 函数声明

func name(parameter-list) (result-list) {
    body
}
*/

func greeting(str1 string) string {
    return "hello " + str1
}

func main() {
    fmt.Println(greeting("张岩"))
}

// 输出结果为:
hello 张岩
```

package main

import "fmt"

/*
// 函数声明

    body
}
*/

// 第一种
func add(x int, y int) int {
    return x + y
}

// 第二种
func sub(x, y int) (z int) {
    z = x - y
    return
}

// 第三种
func first(x int, _ int) int {
    return x
} 

func main() {
    fmt.Println(greeting("张岩"))
}
