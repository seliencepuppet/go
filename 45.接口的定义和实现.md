定义一个叫做 Retriever 的实现接口实现方法

```go
package mock

type Retriever struct {
    Contents string    
}

func (r Retriever) Get(url string) string {
    return r.Contents
}
```

在定义一个指针类型

```go
package real

import (
    "time"
    "net/http"
    "net/http/httputil"
)

type Retriever struct {
    UserAgent string
    TimeOut time.Duration
}

func(r *Retriever) Get(url string) string {
    resp, err := http.Get(url)
    if err != nil {
        panic(err)
    }
    
    result, err := httputil.DumpResponse(resp, true)
    resp.Body.Close()
    
    if err != nil {
        panic(err)
    }
    
    return string(result)
}
```
