
```go
package main

import "fmt"

/*
映射 (Map)

Map是一种哈希表结构, 是一个key/value序列, 每一个key都是不同的, 而且是无序的
在Map中检索数据的时间复杂度是一个常数
*/

func main() {
    ages := map[string]int {
        "Mike":35,
        "John":23,
    }
    fmt.Println(ages)
}
```

第二种创建map对象的函数

```go
package main

import "fmt"

func main() {
    ages1 := make(map[string]int)
    ages1["Mike"] = 20
    ages1["John"] = 12
    fmt.Println(ages1)
    fmt.Println(ages1["Mike"])
}
```

使用 delete 删除map当中的key值

```go
package main

import "fmt"

func main() {
    ages1 := make(map[string]int)
    ages1["Mike"] = 20
    ages1["John"] = 12

    delete(ages1, "Mike")
    delete(ages1, "MIke")
}
```

迭代map的方法

```go
package main

func main() {
    ages1 := make(map[string]int)
    ages1["Mike"] = 20
    ages1["John"] = 12

    for name, age := range ages1 {
        fmt.Println(name, age)
    }
}
```

map类型的零值是 nil, 表示不指向任何哈希表

```go
package main

func main() {
    ages = nil
    fmt.Println(ages == nil)
    fmt.Println(len(ages))
}
```

删除元素

```go
var m map[string]string
fmt.Println("Deleting values")
name, ok := m["name"]
fmt.Println(name, ok)

delete(m "name")
name, ok = m["name"]
fmt.Println(name, ok)
```
