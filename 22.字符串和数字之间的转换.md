
```go
package main
import (
    "fmt"
    "reflect"
)

func main() {
    // 将整数转换为字符串
    // 方法一
    x := 123
    y := fmt.Sprintf("%d", x)
    fmt.Println(x)
    fmt.Println(reflect.TypeOf(y))
}

// 输出结果为:
123
string
```
